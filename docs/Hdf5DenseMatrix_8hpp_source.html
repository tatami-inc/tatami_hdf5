185 // Turn off HDF5's caching, as we'll be handling that. This allows us
186 // to parallelize extractions without locking when the data has already

200 historian.reset(new LruCache);

224 template<bool accrow_, typename ExtractType_>
225 static void extract_base(Index_ primary_start, Index_ primary_length, Value_* target, const ExtractType_& extract_value, Index_ extract_length, Workspace<accrow_>& work) {
226 hsize_t offset[2];
227 hsize_t count[2];
228
229 constexpr int dimdex = (accrow_ != transpose_);
230 offset[1-dimdex] = primary_start;
231 count[1-dimdex] = primary_length;
232
233 constexpr bool indexed = std::is_same<ExtractType_, std::vector<Index_> >::value;
234
235 if constexpr(indexed) {
236 // Take slices across the current chunk for each index. This should be okay if consecutive,
237 // but hopefully they've fixed the problem with non-consecutive slices in:
238 // https://forum.hdfgroup.org/t/union-of-non-consecutive-hyperslabs-is-very-slow/5062
239 count[dimdex] = 1;
240 work.dataspace.selectNone();
241 for (auto idx : extract_value) {
242 offset[dimdex] = idx;
243 work.dataspace.selectHyperslab(H5S_SELECT_OR, count, offset);
244 }
245 count[dimdex] = extract_length; // for the memspace setter.
246 } else {
247 offset[dimdex] = extract_value;
248 count[dimdex] = extract_length;
249 work.dataspace.selectHyperslab(H5S_SELECT_SET, count, offset);
250 }
251
252 // HDF5 is a lot faster when the memspace and dataspace match in dimensionality.
253 // Presumably there is some shuffling that happens inside when dimensions don't match.
254 work.memspace.setExtentSimple(2, count);
255 work.memspace.selectAll();
256
257 work.dataset.read(target, define_mem_type<Value_>(), work.memspace, work.dataspace);
258 }

261 static Index_ extract_chunk(Index_ chunk_id, Index_ dim, Index_ chunk_dim, Value_* target, const ExtractType_& extract_value, Index_ extract_length, Workspace<accrow_>& work) {
262 Index_ chunk_start = chunk_id * chunk_dim;
263 Index_ chunk_end = std::min(dim, chunk_start + chunk_dim);
264 Index_ chunk_actual = chunk_end - chunk_start;
265 extract_base<accrow_>(chunk_start, chunk_actual, target, extract_value, extract_length, work);
266 return chunk_actual;
267 }

269 static void transpose(std::vector<Value_>& cache, std::vector<Value_>& buffer, Index_ actual_dim, Index_ extract_length) {
270 buffer.resize(cache.size());
271 auto output = buffer.begin();
272 for (Index_ x = 0; x < actual_dim; ++x, output += extract_length) {
273 auto in = cache.begin() + x;
274 for (Index_ y = 0; y < extract_length; ++y, in += actual_dim) {
275 *(output + y) = *in;
276 }
277 }
278 cache.swap(buffer);
279 return;
280 }

303 template<bool accrow_, typename ExtractType_>
304 const Value_* extract_with_oracle(Index_ mydim, Index_ chunk_mydim, const ExtractType_& extract_value, Index_ extract_length, Workspace<accrow_>& work) const {
305 auto& pred = *work.futurist;
306 if (pred.predictions_made.size() > pred.predictions_fulfilled) {
307 const auto& chosen = pred.predictions_made[pred.predictions_fulfilled++];
308 return pred.cache_data[chosen.first].data() + extract_length * chosen.second;
309 }
310
311 /* We use the oracle to batch together multiple HDF5 library
312 * calls so that we don't have to reenter the critical section
313 * for every fetch() call. Extracted data are cached for
314 * easy parallel retrieval without involving the HDF5 library.
315 *
316 * We don't bother trying to bundle everything into a single
317 * HDF5 call, as this is a pain to pull data out into the
318 * individual cache buffers afterwards.
319 */
320 Index_ num_chunks = work.num_chunks;
321 Index_ used = 0;
322
323 bool starting = pred.cache_data.empty();
324 if (starting) {
325 pred.cache_data.resize(num_chunks);
326 pred.next_cache_data.resize(num_chunks, std::vector<Value_>(1)); // using a placeholder value for availability checks.
327 pred.chunks_in_need.reserve(num_chunks);
328 } else {
329 pred.next_cache_exists.clear();
330 pred.chunks_in_need.clear();
331 }
332
333 size_t max_predictions = static_cast<size_t>(num_chunks) * chunk_mydim * 2; // double the cache size, basically.
334 pred.predictions_made.clear();
335 pred.predictions_made.reserve(max_predictions);
336
337 for (size_t p = 0; p < max_predictions; ++p) {
338 Index_ current;
339 if (!pred.prediction_stream.next(current)) {
340 break;
341 }
342
343 auto curchunk = current / chunk_mydim;
344 auto curindex = current % chunk_mydim;
345
346 auto it = pred.next_cache_exists.find(curchunk);
347 if (it == pred.next_cache_exists.end()) {
348 pred.next_cache_exists[curchunk] = used;
349 pred.predictions_made.emplace_back(used, curindex);
350
351 auto it2 = pred.cache_exists.find(curchunk);
352 if (it2 != pred.cache_exists.end()) {
353 pred.next_cache_data[used].swap(pred.cache_data[it2->second]);
354 } else {
355 pred.chunks_in_need.emplace_back(curchunk, used);
356 }
357
358 ++used;
359 if (used == num_chunks) {
360 break;
361 }
362 } else {
363 pred.predictions_made.emplace_back(it->second, curindex);
364 }
365 }
366
379 size_t search = 0;
380 for (const auto& c : pred.chunks_in_need) {
381 // If we're lucky enough to already be sitting on an allocation
382 // (e.g., if the last call to this function didn't consume all
383 // cache elements), we just use it directly.
384 if (pred.next_cache_data[c.second].empty()) {
385 while (pred.cache_data[search].empty()) {
386 ++search;
387 }
388
389#ifdef DEBUG
390 // This should never reach the end, as there should always be
391 // 'num_chunks' non-empty elements across cache_data and
392 // next_cache_data. Nonetheless, we add an assertion here.
393 if (search >= pred.cache_data.size()) {
394 throw std::runtime_error("internal cache management error for Hdf5DenseMatrix with oracles");
395 }
396#endif
397
398 pred.next_cache_data[c.second].swap(pred.cache_data[search]);
399 ++search;
400 }
401
402 pred.next_cache_data[c.second].resize(work.per_cache_size); // eventually no-op when all available caches are of the right size.
403 }
404
405 if constexpr(accrow_ == transpose_) {
406 pred.cache_transpose_info.clear();
407 }
408
409#ifndef TATAMI_HDF5_PARALLEL_LOCK
410 #pragma omp critical
411 {
412#else
413 TATAMI_HDF5_PARALLEL_LOCK([&]() -> void {
414#endif
415
416 for (const auto& c : pred.chunks_in_need) {
417 auto& cache_target = pred.next_cache_data[c.second];
418 auto actual_dim = extract_chunk<accrow_>(c.first, mydim, chunk_mydim, cache_target.data(), extract_value, extract_length, work);
419 if constexpr(accrow_ == transpose_) {
420 pred.cache_transpose_info.emplace_back(c.second, actual_dim);
421 }
422 }
423
424#ifndef TATAMI_HDF5_PARALLEL_LOCK
425 }
426#else
427 });
428#endif
429
430 // Applying transpositions to all cached buffers for easier retrieval, but only once the lock is released.
431 if constexpr(accrow_ == transpose_) {
432 for (const auto& x : pred.cache_transpose_info) {
433 transpose(pred.next_cache_data[x.first], work.transposition_buffer, x.second, extract_length);
434 }
435 }
436
437 pred.cache_data.swap(pred.next_cache_data);
438 pred.cache_exists.swap(pred.next_cache_exists);
439
440 pred.predictions_fulfilled = 1; // well, because we just used one.
441 const auto& chosen = pred.predictions_made.front(); // assuming at least one prediction was made, otherwise, why was fetch() even called?
442 return pred.cache_data[chosen.first].data() + extract_length * chosen.second;
443 }

445 template<bool accrow_, typename ExtractType_>
446 const Value_* extract_without_oracle(Index_ i, Index_ mydim, Index_ chunk_mydim, const ExtractType_& extract_value, Index_ extract_length, Workspace<accrow_>& work) const {
447 auto chunk = i / chunk_mydim;
448 auto index = i % chunk_mydim;
449
450 auto& historian = *(work.historian);
451 auto it = historian.cache_exists.find(chunk);
452 if (it != historian.cache_exists.end()) {
453 auto chosen = it->second;
454 historian.cache_data.splice(historian.cache_data.end(), historian.cache_data, chosen); // move to end.
455 return chosen->first.data() + index * extract_length;
456 }
457
458 // Adding a new last element, or recycling the front to the back.
459 typename std::list<typename LruCache::Element>::iterator location;
460 if (historian.cache_data.size() < static_cast<size_t>(work.num_chunks)) {
461 historian.cache_data.emplace_back(std::vector<Value_>(work.per_cache_size), chunk);
462 location = std::prev(historian.cache_data.end());
463 } else {
464 location = historian.cache_data.begin();
465 historian.cache_exists.erase(location->second);
466 location->second = chunk;
467 historian.cache_data.splice(historian.cache_data.end(), historian.cache_data, location); // move to end.
468 }
469 historian.cache_exists[chunk] = location;
470
471 auto& cache_target = location->first;
472 Index_ actual_dim;
473
474#ifndef TATAMI_HDF5_PARALLEL_LOCK
475 #pragma omp critical
476 {
477#else
478 TATAMI_HDF5_PARALLEL_LOCK([&]() -> void {
479#endif
480
481 actual_dim = extract_chunk<accrow_>(chunk, mydim, chunk_mydim, cache_target.data(), extract_value, extract_length, work);
482
483#ifndef TATAMI_HDF5_PARALLEL_LOCK
484 }
485#else
486 });
487#endif
488
489 // Applying a transposition for easier retrieval, but only once the lock is released.
490 if constexpr(accrow_ == transpose_) {
491 transpose(cache_target, work.transposition_buffer, actual_dim, extract_length);
492 }
493
494 return cache_target.data() + index * extract_length;
495 }

497 template<bool accrow_, typename ExtractType_>
498 const Value_* extract(Index_ i, Value_* buffer, const ExtractType_& extract_value, Index_ extract_length, Workspace<accrow_>& work) const {
499 // If there isn't any space for caching, we just extract directly.
500 if (work.num_chunks == 0) {
501 return extract_without_cache(i, buffer, extract_value, extract_length, work);
502 }
503
504 Index_ chunk_mydim, mydim;
505 if constexpr(accrow_ != transpose_) {
506 chunk_mydim = chunk_firstdim;
507 mydim = firstdim;
508 } else {
509 chunk_mydim = chunk_seconddim;
510 mydim = seconddim;
511 }
512
513 const Value_* cache;
514 if (work.futurist) {
515 cache = extract_with_oracle(mydim, chunk_mydim, extract_value, extract_length, work);
516 } else {
517 cache = extract_without_oracle(i, mydim, chunk_mydim, extract_value, extract_length, work);;
518 }
519
520 std::copy(cache, cache + extract_length, buffer);
521 return buffer;
522 }
