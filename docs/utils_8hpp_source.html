101template<bool integer_only, class GroupLike>
102H5::DataSet open_and_check_dataset(const GroupLike& handle, const std::string& name) {
103 // Avoid throwing H5 exceptions.
104 if (!H5Lexists(handle.getId(), name.c_str(), H5P_DEFAULT) || handle.childObjType(name) != H5O_TYPE_DATASET) {
105 throw std::runtime_error("no child dataset named '" + name + "'");
106 }
107
108 auto dhandle = handle.openDataSet(name);
109 auto type = dhandle.getTypeClass();
110 if constexpr(integer_only) {
111 if (type != H5T_INTEGER) {
112 throw std::runtime_error(std::string("expected integer values in the '") + name + "' dataset");
113 }
114 } else {
115 if (type != H5T_INTEGER && type != H5T_FLOAT) {
116 throw std::runtime_error(std::string("expected numeric values in the '") + name + "' dataset");
117 }
118 }
119
120 return dhandle;
121}
