62 } else if (std::is_same<unsigned long, T>::value) {
63 return H5::PredType::NATIVE_ULONG;
64 } else if (std::is_same<long long, T>::value) {

102template<bool integer_only, class GroupLike>
103H5::DataSet open_and_check_dataset(const GroupLike& handle, const std::string& name) {
104 // Avoid throwing H5 exceptions.
105 if (!H5Lexists(handle.getId(), name.c_str(), H5P_DEFAULT) || handle.childObjType(name) != H5O_TYPE_DATASET) {
106 throw std::runtime_error("no child dataset named '" + name + "'");
107 }
108
109 auto dhandle = handle.openDataSet(name);
110 auto type = dhandle.getTypeClass();
111 if constexpr(integer_only) {
112 if (type != H5T_INTEGER) {
113 throw std::runtime_error(std::string("expected integer values in the '") + name + "' dataset");
114 }
115 } else {
116 if (type != H5T_INTEGER && type != H5T_FLOAT) {
117 throw std::runtime_error(std::string("expected numeric values in the '") + name + "' dataset");
118 }
119 }
120
121 return dhandle;
122}

124template<int N>
125std::array<hsize_t, N> get_array_dimensions(const H5::DataSet& handle, const std::string& name) {
126 auto space = handle.getSpace();
127
128 int ndim = space.getSimpleExtentNdims();
129 if (ndim != N) {
130 throw std::runtime_error(std::string("'") + name + "' should be a " + std::to_string(N) + "-dimensional array");
131 }
132
133 std::array<hsize_t, N> dims_out;
134 space.getSimpleExtentDims(dims_out.data(), NULL);
135 return dims_out;
136}
