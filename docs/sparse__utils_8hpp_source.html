11// If we're only interested in the block [X, Y), we don't need to extract the entire set of indices for a primary element.
12// We only need to extract the first Y elements, as all subsequent elements must have indices greater than or equal to Y.
13// Similarly, we only need to extract the last 'dim - X' elements, as all preceding elements must have indices less than X.
14// This allows us to narrow the range of indices to be extracted.
15template<typename Pointer_, typename Index_>
16std::pair<Pointer_, Pointer_> narrow_primary_extraction_range(
17 const Pointer_ ptr_start,
18 const Pointer_ ptr_end,
19 const Index_ block_start,
20 const Index_ block_end,
21 const Index_ secondary_dim
22) {
23 const Pointer_ num_nonzeros = ptr_end - ptr_start;
24 const Pointer_ new_ptr_end = ptr_start + sanisizer::min(block_end, num_nonzeros);
25 const Pointer_ new_ptr_start = ptr_end - sanisizer::min(secondary_dim - block_start, num_nonzeros);
26
27 // It is guaranteed that new_ptr_start <= new_ptr_end.
28 // This is because the difference boils down to:
29 //
30 // -num_nonzeros + num_nonzeros + num_nonzeros, OR
31 // -num_nonzeros + block_end + num_nonzeros, OR
32 // -num_nonzeros + num_nonzeros + secondary_dim - block_start, OR
33 // -num_nonzeros + block_end + secondary_dim - block_start
34 //
35 // All of which are guaranteed to be non-negative.
36 return std::make_pair(new_ptr_start, new_ptr_end);
37}
