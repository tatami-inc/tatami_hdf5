176void write_sparse_matrix_to_hdf5(const tatami::Matrix<Value_, Index_>* mat, H5::Group& location, const WriteSparseMatrixToHdf5Parameters& params) {
177 // Firstly, figuring out the number of non-zero elements, and the maximum value of the indices and data.
178 Value_ lower_data = 0, upper_data = 0;
179 Index_ upper_index = 0;
180 size_t NR = mat->nrow(), NC = mat->ncol();
181 hsize_t non_zeros = 0;
182 bool non_integer = false;
183
184 auto update_stats = [&](const tatami::SparseRange<Value_, Index_>& extracted) -> void {
185 non_zeros += extracted.number;
186 for (size_t i = 0; i < extracted.number; ++i) {
187 auto val = extracted.value[i];
188 if constexpr(!std::is_integral<Value_>::value) {
189 if (std::trunc(val) != val || std::isinf(val)) {
190 non_integer = true;
191 }
192 }
193
194 if (val < lower_data) {
195 lower_data = val;
196 } else if (val > upper_data) {
197 upper_data = val;
198 }
199
200 auto idx = extracted.index[i];
201 if (idx > upper_index) {
202 upper_index = idx;
203 }
204 }
205 };
206
207 if (mat->prefer_rows()) {
208 auto wrk = mat->sparse_row();
209 std::vector<Value_> xbuffer(NC);
210 std::vector<Index_> ibuffer(NC);
211 for (size_t r = 0; r < NR; ++r) {
212 auto extracted = wrk->fetch(r, xbuffer.data(), ibuffer.data());
213 update_stats(extracted);
214 }
215 } else {
216 auto wrk = mat->sparse_column();
217 std::vector<Value_> xbuffer(NR);
218 std::vector<Index_> ibuffer(NR);
219 for (size_t c = 0; c < NC; ++c) {
220 auto extracted = wrk->fetch(c, xbuffer.data(), ibuffer.data());
221 update_stats(extracted);
222 }
223 }
224
225 // Choosing the types.
226 auto data_type = params.data_type;
227 if (data_type == WriteSparseMatrixToHdf5Parameters::StorageType::AUTOMATIC) {
228 if (non_integer && !params.force_integer) {
229 data_type = WriteSparseMatrixToHdf5Parameters::StorageType::DOUBLE;
230 } else {
231 if (lower_data < 0) {
232 if (fits_lower_limit<int8_t>(lower_data) && fits_upper_limit<int8_t>(upper_data)) {
233 data_type = WriteSparseMatrixToHdf5Parameters::StorageType::INT8;
234 } else if (fits_lower_limit<int16_t>(lower_data) && fits_upper_limit<int16_t>(upper_data)) {
235 data_type = WriteSparseMatrixToHdf5Parameters::StorageType::INT16;
236 } else {
237 data_type = WriteSparseMatrixToHdf5Parameters::StorageType::INT32;
238 }
239 } else {
240 if (fits_upper_limit<uint8_t>(upper_data)) {
241 data_type = WriteSparseMatrixToHdf5Parameters::StorageType::UINT8;
242 } else if (fits_upper_limit<uint16_t>(upper_data)) {
243 data_type = WriteSparseMatrixToHdf5Parameters::StorageType::UINT16;
244 } else {
245 data_type = WriteSparseMatrixToHdf5Parameters::StorageType::UINT32;
246 }
247 }
248 }
249 }
250
251 auto index_type = params.index_type;
252 if (index_type == WriteSparseMatrixToHdf5Parameters::StorageType::AUTOMATIC) {
253 if (fits_upper_limit<uint8_t>(upper_index)) {
254 index_type = WriteSparseMatrixToHdf5Parameters::StorageType::UINT8;
255 } else if (fits_upper_limit<uint16_t>(upper_index)) {
256 index_type = WriteSparseMatrixToHdf5Parameters::StorageType::UINT16;
257 } else {
258 index_type = WriteSparseMatrixToHdf5Parameters::StorageType::UINT32;
259 }
260 }
261
262 // And then saving it. This time we have no choice but to iterate by the desired dimension.
263 H5::DataSet data_ds = create_1d_compressed_hdf5_dataset(location, data_type, params.data_name, non_zeros, params.deflate_level, params.chunk_size);
264 H5::DataSet index_ds = create_1d_compressed_hdf5_dataset(location, index_type, params.index_name, non_zeros, params.deflate_level, params.chunk_size);
265 hsize_t count = 0, offset = 0;
266 H5::DataSpace inspace(1, &non_zeros);
267 H5::DataSpace outspace(1, &non_zeros);
268 const auto& dstype = define_mem_type<Value_>();
269 const auto& ixtype = define_mem_type<Index_>();
270
271 auto fill_datasets = [&](const tatami::SparseRange<Value_, Index_>& extracted) -> void {
272 count = extracted.number;
273 if (count) {
274 inspace.setExtentSimple(1, &count);
275 outspace.selectHyperslab(H5S_SELECT_SET, &count, &offset);
276 data_ds.write(extracted.value, dstype, inspace, outspace);
277 index_ds.write(extracted.index, ixtype, inspace, outspace);
278 offset += count;
279 }
280 };
281
282 auto layout = params.columnar;
283 if (layout == WriteSparseMatrixToHdf5Parameters::StorageLayout::AUTOMATIC) {
284 if (mat->prefer_rows()) {
285 layout = WriteSparseMatrixToHdf5Parameters::StorageLayout::ROW;
286 } else {
287 layout = WriteSparseMatrixToHdf5Parameters::StorageLayout::COLUMN;
288 }
289 }
290
291 std::vector<hsize_t> ptrs;
292 if (layout == WriteSparseMatrixToHdf5Parameters::StorageLayout::ROW) {
293 ptrs.resize(NR + 1);
294 auto wrk = mat->sparse_row();
295 std::vector<Value_> xbuffer(NC);
296 std::vector<Index_> ibuffer(NC);
297 for (size_t r = 0; r < NR; ++r) {
298 auto extracted = wrk->fetch(r, xbuffer.data(), ibuffer.data());
299 fill_datasets(extracted);
300 ptrs[r+1] = ptrs[r] + extracted.number;
301 }
302 } else {
303 ptrs.resize(NC + 1);
304 auto wrk = mat->sparse_column();
305 std::vector<Value_> xbuffer(NR);
306 std::vector<Index_> ibuffer(NR);
307 for (size_t c = 0; c < NC; ++c) {
308 auto extracted = wrk->fetch(c, xbuffer.data(), ibuffer.data());
309 fill_datasets(extracted);
310 ptrs[c+1] = ptrs[c] + extracted.number;
311 }
312 }
313
314 // Saving the pointers.
315 hsize_t ptr_len = ptrs.size();
316 H5::DataSet ptr_ds = create_1d_compressed_hdf5_dataset(location, H5::PredType::NATIVE_HSIZE, params.ptr_name, ptr_len, params.deflate_level, params.chunk_size);
317 H5::DataSpace ptr_space(1, &ptr_len);
318 ptr_ds.write(ptrs.data(), H5::PredType::NATIVE_HSIZE, ptr_space);
319
320 return;
321}
