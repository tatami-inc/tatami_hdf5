117WriteStorageType choose_index_type(const std::optional<WriteStorageType>& index_type, Index_ upper_index) {
118 if (!index_type.has_value()) {
119 if (fits_upper_limit<std::uint8_t>(upper_index)) {
120 return WriteStorageType::UINT8;
121 } else if (fits_upper_limit<std::uint16_t>(upper_index)) {
122 return WriteStorageType::UINT16;
123 } else if (fits_upper_limit<std::uint32_t>(upper_index)) {
124 return WriteStorageType::UINT32;
125 } else if (fits_upper_limit<std::uint64_t>(upper_index)) {
126 return WriteStorageType::UINT64;
127 }
128 throw std::runtime_error("no type can store the largest index");
129 }
130
131 const auto itype = *index_type;
132 if (
133 (itype == WriteStorageType::INT8 && !fits_upper_limit<std::int8_t >(upper_index)) ||
134 (itype == WriteStorageType::UINT8 && !fits_upper_limit<std::uint8_t >(upper_index)) ||
135 (itype == WriteStorageType::INT16 && !fits_upper_limit<std::int16_t >(upper_index)) ||
136 (itype == WriteStorageType::UINT16 && !fits_upper_limit<std::uint16_t>(upper_index)) ||
137 (itype == WriteStorageType::INT32 && !fits_upper_limit<std::int32_t >(upper_index)) ||
138 (itype == WriteStorageType::UINT32 && !fits_upper_limit<std::uint32_t>(upper_index)) ||
139 (itype == WriteStorageType::INT64 && !fits_upper_limit<std::int64_t >(upper_index)) ||
140 (itype == WriteStorageType::UINT64 && !fits_upper_limit<std::uint64_t>(upper_index)) ||
141 (itype == WriteStorageType::DOUBLE /* must be integer */ ) ||
142 (itype == WriteStorageType::FLOAT /* must be integer */ )
143 ) {
144 throw std::runtime_error("specified type cannot store the largest index");
145 }

180template<typename Value_, typename Index_>
181struct WriteSparseHdf5Statistics {
182 Value_ lower_data = 0;
183 Value_ upper_data = 0;
184 Index_ upper_index = 0;
185 hsize_t non_zeros = 0;
186 bool has_decimal = false;
187 bool has_nonfinite = false;
188
189 void add_value(Value_ val) {
190 if constexpr(!std::is_integral<Value_>::value) {
191 if (std::trunc(val) != val) {
192 has_decimal = true;
193 }
194 if (!std::isfinite(val)) {
195 has_nonfinite = true;
196 }
197 }
198
199 if (val < lower_data) {
200 lower_data = val;
201 } else if (val > upper_data) {
202 upper_data = val;
203 }
204 }
205
206 void add_index(Index_ idx) {
207 if (idx > upper_index) {
208 upper_index = idx;
209 }
210 }
211};

213template<typename Value_, typename Index_>
214void update_hdf5_stats(const tatami::SparseRange<Value_, Index_>& extracted, WriteSparseHdf5Statistics<Value_, Index_>& output) {
215 // We need to protect the addition just in case it overflows from having too many non-zero elements.
216 output.non_zeros = sanisizer::sum<hsize_t>(output.non_zeros, extracted.number);
217
218 for (Index_ i = 0; i < extracted.number; ++i) {
219 output.add_value(extracted.value[i]);
220 }
221
222 for (Index_ i = 0; i < extracted.number; ++i) {
223 output.add_index(extracted.index[i]);
224 }
225}

227template<typename Value_, typename Index_>
228void update_hdf5_stats(const Value_* extracted, Index_ n, WriteSparseHdf5Statistics<Value_, Index_>& output) {
229 Index_ local_nonzero = 0;
230 for (Index_ i = 0; i < n; ++i) {
231 auto val = extracted[i];
232 if (val == 0) {
233 continue;
234 }
235 ++local_nonzero;
236 output.add_value(val);
237 output.add_index(i);
238 }
239
240 // Checking that there aren't overflows, but doing so outside of the hot loop for perf.
241 output.non_zeros = sanisizer::sum<hsize_t>(output.non_zeros, local_nonzero);
242}

348template<typename Value_, typename Index_>
349void write_compressed_sparse_matrix(const tatami::Matrix<Value_, Index_>& mat, H5::Group& location, const WriteCompressedSparseMatrixOptions& params) {
350 auto stats = write_sparse_hdf5_statistics(mat, params.num_threads);
351 const auto data_type = choose_data_type(params.data_type, stats.lower_data, stats.upper_data, stats.has_decimal, params.force_integer, stats.has_nonfinite);
352 const auto index_type = choose_index_type(params.index_type, stats.upper_index);
353
354 // Choosing the layout.
355 WriteStorageLayout layout;
356 if (params.columnar.has_value()) {
357 layout = *(params.columnar);
358 } else {
359 if (mat.prefer_rows()) {
360 layout = WriteStorageLayout::ROW;
361 } else {
362 layout = WriteStorageLayout::COLUMN;
363 }
364 }
365
366 // Choosing the names.
367 std::string data_name;
368 if (params.data_name.has_value()) {
369 data_name = *(params.data_name);
370 } else {
371 data_name = "data";
372 }
373
374 std::string index_name;
375 if (params.index_name.has_value()) {
376 index_name = *(params.index_name);
377 } else {
378 index_name = "indices";
379 }
380
381 std::string ptr_name;
382 if (params.ptr_name.has_value()) {
383 ptr_name = *(params.ptr_name);
384 } else {
385 ptr_name = "indptr";
386 }
387
388 // And then saving it. This time we have no choice but to iterate by the desired dimension.
389 auto non_zeros = stats.non_zeros;
390 H5::DataSet data_ds = create_1d_compressed_hdf5_dataset(location, data_type, data_name, non_zeros, params.deflate_level, params.chunk_size);
391 H5::DataSet index_ds = create_1d_compressed_hdf5_dataset(location, index_type, index_name, non_zeros, params.deflate_level, params.chunk_size);
392 hsize_t offset = 0;
393 H5::DataSpace inspace(1, &non_zeros);
394 H5::DataSpace outspace(1, &non_zeros);
395 const auto& dstype = define_mem_type<Value_>();
396 const auto& ixtype = define_mem_type<Index_>();
397
398 Index_ NR = mat.nrow(), NC = mat.ncol();
399 std::vector<hsize_t> ptrs;
400
401 auto fill_datasets = [&](const Value_* vptr, const Index_* iptr, hsize_t count) -> void {
402 if (count) {
403 inspace.setExtentSimple(1, &count);
404 outspace.selectHyperslab(H5S_SELECT_SET, &count, &offset);
405 data_ds.write(vptr, dstype, inspace, outspace);
406 index_ds.write(iptr, ixtype, inspace, outspace);
407 offset += count;
408 }
409 };
410
411 if (mat.sparse()) {
412 if (layout == WriteStorageLayout::ROW) {
413 ptrs.resize(sanisizer::sum<decltype(ptrs.size())>(NR, 1));
414 auto xbuffer = tatami::create_container_of_Index_size<std::vector<Value_> >(NC);
415 auto ibuffer = tatami::create_container_of_Index_size<std::vector<Index_> >(NC);
416
417 auto wrk = tatami::consecutive_extractor<true>(mat, true, static_cast<Index_>(0), NR);
418 for (Index_ r = 0; r < NR; ++r) {
419 auto extracted = wrk->fetch(r, xbuffer.data(), ibuffer.data());
420 fill_datasets(extracted.value, extracted.index, extracted.number);
421 ptrs[r+1] = ptrs[r] + extracted.number;
422 }
423
424 } else {
425 ptrs.resize(sanisizer::sum<decltype(ptrs.size())>(NC, 1));
426 auto xbuffer = tatami::create_container_of_Index_size<std::vector<Value_> >(NR);
427 auto ibuffer = tatami::create_container_of_Index_size<std::vector<Index_> >(NR);
428
429 auto wrk = tatami::consecutive_extractor<true>(mat, false, static_cast<Index_>(0), NC);
430 for (Index_ c = 0; c < NC; ++c) {
431 auto extracted = wrk->fetch(c, xbuffer.data(), ibuffer.data());
432 fill_datasets(extracted.value, extracted.index, extracted.number);
433 ptrs[c+1] = ptrs[c] + extracted.number;
434 }
435 }
436
437 } else {
438 std::vector<Value_> sparse_xbuffer;
439 std::vector<Index_> sparse_ibuffer;
440 auto fill_datasets_from_dense = [&](const Value_* extracted, Index_ n) -> hsize_t {
441 sparse_xbuffer.clear();
442 sparse_ibuffer.clear();
443 for (Index_ i = 0; i < n; ++i) {
444 if (extracted[i]) {
445 sparse_xbuffer.push_back(extracted[i]);
446 sparse_ibuffer.push_back(i);
447 }
448 }
449
450 hsize_t count = sparse_xbuffer.size();
451 fill_datasets(sparse_xbuffer.data(), sparse_ibuffer.data(), count);
452 return count;
453 };
454
455 if (layout == WriteStorageLayout::ROW) {
456 ptrs.resize(sanisizer::sum<decltype(ptrs.size())>(NR, 1));
457 auto dbuffer = tatami::create_container_of_Index_size<std::vector<Value_> >(NC);
458 auto wrk = tatami::consecutive_extractor<false>(mat, true, static_cast<Index_>(0), NR);
459 for (Index_ r = 0; r < NR; ++r) {
460 auto extracted = wrk->fetch(r, dbuffer.data());
461 auto count = fill_datasets_from_dense(extracted, NC);
462 ptrs[r+1] = ptrs[r] + count;
463 }
464
465 } else {
466 ptrs.resize(sanisizer::sum<decltype(ptrs.size())>(NC, 1));
467 auto dbuffer = tatami::create_container_of_Index_size<std::vector<Value_> >(NR);
468 auto wrk = tatami::consecutive_extractor<false>(mat, false, static_cast<Index_>(0), NC);
469 for (Index_ c = 0; c < NC; ++c) {
470 auto extracted = wrk->fetch(c, dbuffer.data());
471 auto count = fill_datasets_from_dense(extracted, NR);
472 ptrs[c+1] = ptrs[c] + count;
473 }
474 }
475 }
476
477 // Saving the pointers.
478 auto ptr_len = sanisizer::cast<hsize_t>(ptrs.size());
479 H5::DataSet ptr_ds = create_1d_compressed_hdf5_dataset(
480 location,
481 choose_ptr_type(params.ptr_type, ptrs.back()),
482 ptr_name,
483 ptr_len,
484 params.deflate_level,
485 params.chunk_size
486 );
487 H5::DataSpace ptr_space(1, &ptr_len);
488 ptr_ds.write(ptrs.data(), H5::PredType::NATIVE_HSIZE, ptr_space);
489
490 return;
491}
