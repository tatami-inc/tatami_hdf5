47tatami::CompressedSparseMatrix<ROW, T, IDX, U, V, W> load_hdf5_compressed_sparse_matrix(
48 size_t nr,
49 size_t nc,
50 const std::string& file,

52 const std::string& idx,
53 const std::string& ptr)
54{
55 H5::H5File file_handle(file, H5F_ACC_RDONLY);
56
57 auto dhandle = open_and_check_dataset<false>(file_handle, vals);
58 const size_t nonzeros = get_array_dimensions<1>(dhandle, "vals")[0];
59 U x(nonzeros);
60 dhandle.read(x.data(), define_mem_type<typename U::value_type>());
61
62 auto ihandle = open_and_check_dataset<true>(file_handle, idx);
63 if (get_array_dimensions<1>(ihandle, "idx")[0] != nonzeros) {
64 throw std::runtime_error("number of non-zero elements is not consistent between 'data' and 'idx'");
65 }
66 V i(nonzeros);
67 ihandle.read(i.data(), define_mem_type<typename V::value_type>());
68
69 auto phandle = open_and_check_dataset<true>(file_handle, ptr);
70 const size_t ptr_size = get_array_dimensions<1>(phandle, "ptr")[0];
71 if (ptr_size != (ROW ? nr : nc) + 1) {
72 throw std::runtime_error("'ptr' dataset should have length equal to the number of " + (ROW ? std::string("rows") : std::string("columns")) + " plus 1");
73 }
74
75 // Because HDF5 doesn't have a native type for size_t.
76 W p(ptr_size);
77 if constexpr(std::is_same<size_t, typename W::value_type>::value) {
78 if constexpr(std::is_same<size_t, hsize_t>::value) {
79 phandle.read(p.data(), H5::PredType::NATIVE_HSIZE);
80 } else {
81 std::vector<hsize_t> p0(ptr_size);
82 phandle.read(p0.data(), H5::PredType::NATIVE_HSIZE);
83 std::copy(p0.begin(), p0.end(), p.begin());
84 }
85 } else {
86 phandle.read(p.data(), define_mem_type<typename W::value_type>());
87 }
88
89 return tatami::CompressedSparseMatrix<ROW, T, IDX, U, V, W>(nr, nc, std::move(x), std::move(i), std::move(p));
90}
